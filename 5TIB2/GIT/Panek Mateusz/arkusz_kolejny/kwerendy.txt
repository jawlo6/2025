SELECT plik, nr_klienta, liczba_odbitek FROM zamowienia WHERE rodzaj="matowy" AND liczba_odbitek>100;

SELECT zamowienia.liczba_odbitek, zdjecia.cena, zamowienia.liczba_odbitek*zdjecia.cena AS 'Do zapłaty' 
FROM zamowienia, zdjecia WHERE zamowienia.rodzaj=zdjecia.rodzaj AND nr_klienta=3;

SELECT SUM(liczba_odbitek) FROM zamowienia WHERE rodzaj="błyszczący";
